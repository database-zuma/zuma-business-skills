// ==================== BAGIAN 1: SUPABASE SYNC ====================

/**
 * Ambil data dari Supabase dengan pagination
 * Filter: area = 'Jatim'
 */
function fetchAllRawFromSupabase() {
  var endpoint = SUPABASE_URL + '/rest/v1/' + SUPABASE_TABLE_NAME;
  var allData = [];
  var pageSize = 1000;
  var offset = 0;

  while (true) {
    var from = offset;
    var to = offset + pageSize - 1;

    var url = endpoint
      + '?select=*'
      + '&area=eq.' + encodeURIComponent('Jatim');

    var options = {
      method: 'get',
      headers: {
        'apikey': SUPABASE_KEY,
        'Authorization': 'Bearer ' + SUPABASE_KEY,
        'Content-Type': 'application/json',
        'Range-Unit': 'items',
        'Range': from + '-' + to
      },
      muteHttpExceptions: true
    };

    var response = UrlFetchApp.fetch(url, options);
    var statusCode = response.getResponseCode();
    var text = response.getContentText();

    if (statusCode !== 200 && statusCode !== 206) {
      Logger.log('Error response: ' + text);
      throw new Error('Gagal fetch data dari Supabase. HTTP Code: ' + statusCode);
    }

    var pageData = JSON.parse(text);
    Logger.log('Page offset ' + offset + ' - jumlah baris: ' + pageData.length);

    if (!pageData.length) break;

    allData = allData.concat(pageData);

    if (pageData.length < pageSize) break;

    offset += pageSize;
  }

  Logger.log('TOTAL baris raw (area Jatim): ' + allData.length);
  return allData;
}

/**
 * Tulis data Supabase ke sheet
 */
function dumpSupabaseRawToSheet() {
  var rawData = fetchAllRawFromSupabase();
  if (!rawData || rawData.length === 0) {
    Logger.log('Tidak ada data raw dari Supabase (area Jatim).');
    return;
  }

  // Buka spreadsheet target berdasarkan ID
  var targetSpreadsheet;
  try {
    if (TARGET_SPREADSHEET_ID === 'MASUKKAN_SPREADSHEET_ID_DISINI' || !TARGET_SPREADSHEET_ID) {
      // Jika belum diisi, gunakan spreadsheet aktif
      targetSpreadsheet = SpreadsheetApp.getActiveSpreadsheet();
      Logger.log('Menggunakan spreadsheet aktif (ID belum diset)');
    } else {
      targetSpreadsheet = SpreadsheetApp.openById(TARGET_SPREADSHEET_ID);
      Logger.log('Menggunakan target spreadsheet: ' + TARGET_SPREADSHEET_ID);
    }
  } catch (e) {
    throw new Error('Gagal membuka spreadsheet target. Pastikan ID benar dan Anda punya akses. Error: ' + e.message);
  }

  var sheet = targetSpreadsheet.getSheetByName(TARGET_SHEET_NAME);
  if (!sheet) {
    sheet = targetSpreadsheet.insertSheet(TARGET_SHEET_NAME);
    Logger.log('Sheet "' + TARGET_SHEET_NAME + '" dibuat baru');
  }

  sheet.clearContents();
  sheet.getRange(1, 1, 1, COLUMN_ORDER.length).setValues([COLUMN_ORDER]);

  var rows = rawData.map(function (row) {
    return COLUMN_ORDER.map(function (colName) {
      var value = row[colName];
      
      if (colName === 'size') {
        if (value === undefined || value === null || value === '') return '';
        return "'" + value.toString();
      }
      
      return value !== undefined ? value : '';
    });
  });

  sheet.getRange(2, 1, rows.length, COLUMN_ORDER.length).setValues(rows);

  var sizeColIndex = COLUMN_ORDER.indexOf('size') + 1;
  if (sizeColIndex > 0) {
    var sizeRange = sheet.getRange(2, sizeColIndex, rows.length, 1);
    sizeRange.setNumberFormat('@');
  }

  Logger.log('Berhasil menulis ' + rows.length + ' baris ke sheet "' + TARGET_SHEET_NAME + '" di spreadsheet target');
}

/**
 * Fungsi utama untuk gabungkan Planogram dan Stock
 */
function gabungkanPlanogramStock() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  SpreadsheetApp.getActiveSpreadsheet().toast('Memproses data...', '⏳ Loading', -1);
  
  const planogramSheet = ss.getSheetByName('Planogram');
  const stockSheet = ss.getSheetByName('Stock');
  let hasilSheet = ss.getSheetByName('Hasil Gabungan');
  
  if (!planogramSheet || !stockSheet) {
    SpreadsheetApp.getUi().alert('Error: Sheet "Planogram" atau "Stock" tidak ditemukan!');
    return;
  }
  
  if (!hasilSheet) {
    hasilSheet = ss.insertSheet('Hasil Gabungan');
  } else {
    hasilSheet.clear();
  }
  
  const planogramData = planogramSheet.getDataRange().getValues();
  const planogramHeaders = planogramData[0];
  
  const stockData = stockSheet.getDataRange().getValues();
  const stockHeaders = stockData[0];
  
  console.log('Stock Headers:', stockHeaders);
  console.log('Planogram Headers:', planogramHeaders);
  
  // Index untuk kolom Stock
  const stockIndexes = {};
  stockHeaders.forEach((header, index) => {
    const headerLower = header.toString().toLowerCase().trim();
    
    if (headerLower === 'store name' || headerLower === 'store') {
      stockIndexes.storeName = index;
    } else if (headerLower === 'kode mix') {
      stockIndexes.kodeMix = index;
    } else if (headerLower === 'size') {
      stockIndexes.size = index;
    } else if (headerLower === 'kuantitas' || headerLower === 'quantity' || headerLower === 'qty') {
      stockIndexes.kuantitas = index;
    }
  });
  
  // Fallback ke index tetap jika tidak ditemukan
  if (stockIndexes.storeName === undefined) stockIndexes.storeName = 0;
  if (stockIndexes.kodeMix === undefined) stockIndexes.kodeMix = 1;
  if (stockIndexes.size === undefined) stockIndexes.size = 2;
  if (stockIndexes.kuantitas === undefined) stockIndexes.kuantitas = 3;
  
  console.log('Stock Indexes:', stockIndexes);
  
  // Index untuk kolom Planogram
  let planogramTierIndex = -1;
  let planogramArticleMixIndex = -1;
  
  planogramHeaders.forEach((header, index) => {
    const headerStr = String(header || '');
    const headerLower = headerStr.toLowerCase().trim();
    
    if (headerLower === 'tier') {
      planogramTierIndex = index;
    } else if (headerLower === 'article mix') {
      planogramArticleMixIndex = index;
    }
  });
  
  if (planogramArticleMixIndex === -1) {
    SpreadsheetApp.getUi().alert('Error: Kolom "article mix" tidak ditemukan di sheet Planogram');
    return;
  }
  
  // Buat map untuk stock data
  const stockMap = new Map();
  for (let i = 1; i < stockData.length; i++) {
    const row = stockData[i];
    
    const storeName = row[stockIndexes.storeName] ? String(row[stockIndexes.storeName]).trim() : '';
    const kodeMix = row[stockIndexes.kodeMix] ? String(row[stockIndexes.kodeMix]).trim() : '';
    const size = row[stockIndexes.size] ? String(row[stockIndexes.size]).trim() : '';
    const kuantitas = Number(row[stockIndexes.kuantitas]) || 0;
    
    if (storeName && kodeMix && size) {
      const key = `${storeName}_${kodeMix}_${size}`;
      stockMap.set(key, kuantitas);
      
      if (i <= 5) {
        console.log(`Stock row ${i}: ${key} = ${kuantitas}`);
      }
    }
  }
  
  console.log('Total stock entries loaded:', stockMap.size);
  
  // Identifikasi kolom size di Planogram
  const sizeStartIndex = planogramArticleMixIndex + 1;
  let sizeEndIndex = -1;
  
  for (let i = sizeStartIndex; i < planogramHeaders.length; i++) {
    const header = String(planogramHeaders[i] || '');
    if (header === '' || header.includes('AVG') || header.includes('Sales') || 
        header.includes('Rekomendasi') || header.includes('%') || header.includes('Input')) {
      sizeEndIndex = i - 1;
      break;
    }
  }
  
  if (sizeEndIndex === -1 || sizeEndIndex < sizeStartIndex) {
    sizeEndIndex = Math.min(sizeStartIndex + 20, planogramHeaders.length - 1);
  }
  
  // Ambil list size yang valid
  const sizeColumns = [];
  for (let i = sizeStartIndex; i <= sizeEndIndex; i++) {
    const sizeHeader = String(planogramHeaders[i] || '').trim();
    if (sizeHeader) {
      sizeColumns.push(sizeHeader);
    }
  }
  
  console.log('Size columns found:', sizeColumns);
  
  // Buat header untuk hasil pivot
  const hasilHeader = [
    'Store Name',
    'Article Mix', 
    'Gender',
    'Series',
    'Article',
    'Tier'
  ];
  
  sizeColumns.forEach(size => {
    hasilHeader.push(`${size}_Plan`);
  });
  
  sizeColumns.forEach(size => {
    hasilHeader.push(`${size}_Stock`);
  });
  
  hasilHeader.push('Total_Plan');
  hasilHeader.push('Total_Stock');
  hasilHeader.push('Count_Plan>0');
  hasilHeader.push('Count_Stock>0');
  hasilHeader.push('Count_Article_Plan');
  hasilHeader.push('Count_Article_Stock');
  
  const hasil = [hasilHeader];
  
  // Process data per row
  for (let i = 1; i < planogramData.length; i++) {
    const row = planogramData[i];
    const storeName = row[0];
    const gender = row[1];
    const series = row[2];
    const article = row[3];
    
    const articleMix = planogramArticleMixIndex !== -1 ? (row[planogramArticleMixIndex] || '') : '';
    const tierFromPlanogram = planogramTierIndex !== -1 ? (row[planogramTierIndex] || '') : '';
    
    if (!storeName || !articleMix) continue;
    
    const newRow = [
      storeName,
      articleMix,
      gender,
      series,
      article,
      tierFromPlanogram
    ];
    
    let totalStock = 0;
    let totalPlan = 0;
    let countPlanPositive = 0;
    let countStockPositive = 0;
    let hasData = false;
    
    const planData = [];
    const stockDataArray = [];
    
    for (let j = 0; j < sizeColumns.length; j++) {
      const size = sizeColumns[j];
      const planogramIndex = sizeStartIndex + j;
      const cellValue = row[planogramIndex];
      
      let stockQty = 0;
      let planQty = 0;
      
      if (cellValue !== undefined && cellValue !== null && cellValue !== '') {
        planQty = Number(cellValue) || 0;
        hasData = true;
        
        const stockKey = `${storeName}_${articleMix}_${size}`;
        stockQty = stockMap.get(stockKey) || 0;
        
        totalStock += stockQty;
        totalPlan += planQty;
        
        if (planQty > 0) {
          countPlanPositive++;
        }
        
        if (planQty > 0 && stockQty > 0) {
          countStockPositive++;
        }
      }
      
      planData.push(planQty || '');
      stockDataArray.push(stockQty || '');
    }
    
    let countArticlePlan = 0;
    let countArticleStock = 0;
    
    if (totalPlan > 0) {
      countArticlePlan = 1;
      if (totalStock > 0) {
        countArticleStock = 1;
      }
    }
    
    newRow.push(...planData);
    newRow.push(...stockDataArray);
    newRow.push(totalPlan || '');
    newRow.push(totalStock || '');
    newRow.push(countPlanPositive);
    newRow.push(countStockPositive);
    newRow.push(countArticlePlan);
    newRow.push(countArticleStock);
    
    if (hasData) {
      hasil.push(newRow);
    }
  }
  
  // Output hasil ke sheet
  if (hasil.length > 1) {
    hasilSheet.getRange(1, 1, hasil.length, hasil[0].length).setValues(hasil);
    
    // Formatting
    const headerRange = hasilSheet.getRange(1, 1, 1, hasil[0].length);
    headerRange.setFontWeight('bold');
    headerRange.setBackground('#4a90e2');
    headerRange.setFontColor('#000000');
    headerRange.setHorizontalAlignment('center');
    
    const planStartCol = 7;
    const planEndCol = 6 + sizeColumns.length;
    hasilSheet.getRange(1, planStartCol, hasil.length, sizeColumns.length).setBackground('#e3f2fd');
    
    const stockStartCol = planEndCol + 1;
    const stockEndCol = stockStartCol + sizeColumns.length - 1;
    hasilSheet.getRange(1, stockStartCol, hasil.length, sizeColumns.length).setBackground('#e8f5e9');
    
    const summaryStartCol = stockEndCol + 1;
    hasilSheet.getRange(1, summaryStartCol, hasil.length, 2).setBackground('#f5f5f5');
    hasilSheet.getRange(1, summaryStartCol, hasil.length, 2).setFontWeight('bold');
    
    const countSizeStartCol = summaryStartCol + 2;
    hasilSheet.getRange(1, countSizeStartCol, hasil.length, 2).setBackground('#fff3e0');
    
    const countArticleStartCol = countSizeStartCol + 2;
    hasilSheet.getRange(1, countArticleStartCol, hasil.length, 2).setBackground('#f3e5f5');
    
    for (let i = 1; i <= hasil[0].length; i++) {
      hasilSheet.autoResizeColumn(i);
    }
    
    hasilSheet.getRange(1, 1, hasil.length, hasil[0].length)
      .setBorder(true, true, true, true, true, true);
    
    if (hasil.length > 1) {
      hasilSheet.getRange(2, 7, hasil.length - 1, hasil[0].length - 6)
        .setNumberFormat('#,##0');
    }
    
    hasilSheet.setFrozenRows(1);
    hasilSheet.setFrozenColumns(6);
    
    SpreadsheetApp.getActiveSpreadsheet().toast(
      `Total baris: ${hasil.length - 1} | Total size: ${sizeColumns.length}`, 
      '✅ Proses Selesai!', 
      5
    );
  } else {
    SpreadsheetApp.getUi().alert('Tidak ada data yang dapat digabungkan.');
  }
}

function updateDailyReport() {
  try {
    SpreadsheetApp.getActiveSpreadsheet().toast('Sedang memproses data...', '⏳ Loading', -1);

/**
 * Process source data dan hitung metrics
 */
function processSourceData(sourceSheet, stockSheet) {
  const sourceData = sourceSheet.getDataRange().getValues();
  const headerRow = sourceData[0];
  
  const columnIndexes = {
    storeNameCol: headerRow.indexOf('Store Name'),
    totalPlanCol: headerRow.indexOf('Total_Plan'),
    totalStockCol: headerRow.indexOf('Total_Stock'),
    countPlanCol: headerRow.indexOf('Count_Plan>0'),
    countStockCol: headerRow.indexOf('Count_Stock>0'),
    countArticlePlanCol: headerRow.indexOf('Count_Article_Plan'),
    countArticleCol: headerRow.indexOf('Count_Article_Stock')
  };
  
  for (const [key, value] of Object.entries(columnIndexes)) {
    if (value === -1) {
      throw new Error(`Kolom tidak ditemukan: ${key}`);
    }
  }
  
  // Process Stock sheet untuk Sell-Through
  let sellThroughMetrics = {};
  if (stockSheet) {
    const stockData = stockSheet.getDataRange().getValues();
    const stockHeaders = stockData[0];
    
    console.log('Stock Headers untuk ST:', stockHeaders);
    
    const stockIndexes = {
      storeNameCol: 0,
      keluarCol: 6,
      saldoAwalCol: 4,
      masukCol: 5
    };
    
    console.log('Kolom Store:', stockHeaders[stockIndexes.storeNameCol]);
    console.log('Kolom Keluar:', stockHeaders[stockIndexes.keluarCol]);
    console.log('Kolom Saldo Awal:', stockHeaders[stockIndexes.saldoAwalCol]);
    console.log('Kolom Masuk:', stockHeaders[stockIndexes.masukCol]);
    
    const storeSTData = {};
    for (let i = 1; i < stockData.length; i++) {
      const row = stockData[i];
      const storeName = row[stockIndexes.storeNameCol];
      
      if (!storeName || storeName === '') continue;
      
      if (!storeSTData[storeName]) {
        storeSTData[storeName] = {
          totalKeluar: 0,
          totalSaldoAwal: 0,
          totalMasuk: 0
        };
      }
      
      const keluar = Number(row[stockIndexes.keluarCol]) || 0;
      const saldoAwal = Number(row[stockIndexes.saldoAwalCol]) || 0;
      const masuk = Number(row[stockIndexes.masukCol]) || 0;
      
      storeSTData[storeName].totalKeluar += keluar;
      storeSTData[storeName].totalSaldoAwal += saldoAwal;
      storeSTData[storeName].totalMasuk += masuk;
    }
    
    console.log('Store ST Data:', storeSTData);
    
    for (const store in storeSTData) {
      const data = storeSTData[store];
      const denominator = data.totalSaldoAwal + data.totalMasuk;
      
      if (denominator > 0) {
        const st = (data.totalKeluar / denominator * 100).toFixed(2);
        sellThroughMetrics[store] = st + '%';
        console.log(`${store}: Keluar=${data.totalKeluar}, SaldoAwal+Masuk=${denominator}, ST=${st}%`);
      } else {
        sellThroughMetrics[store] = '0.00%';
      }
    }
  }
  
  // Calculate metrics FF, FA, FS
  const storeMetrics = {};
  for (let i = 1; i < sourceData.length; i++) {
    const row = sourceData[i];
    const storeName = row[columnIndexes.storeNameCol];
    
    if (!storeName || storeName === '') continue;
    
    if (!storeMetrics[storeName]) {
      storeMetrics[storeName] = {
        totalPlan: 0,
        totalStock: 0,
        countPlan: 0,
        countStock: 0,
        countArticlePlan: 0,
        countArticle: 0
      };
    }
    
    storeMetrics[storeName].totalPlan += row[columnIndexes.totalPlanCol] || 0;
    storeMetrics[storeName].totalStock += row[columnIndexes.totalStockCol] || 0;
    storeMetrics[storeName].countPlan += row[columnIndexes.countPlanCol] || 0;
    storeMetrics[storeName].countStock += row[columnIndexes.countStockCol] || 0;
    storeMetrics[storeName].countArticlePlan += row[columnIndexes.countArticlePlanCol] || 0;
    storeMetrics[storeName].countArticle += row[columnIndexes.countArticleCol] || 0;
  }
  
  const calculatedMetrics = {};
  for (const store in storeMetrics) {
    const metrics = storeMetrics[store];
    calculatedMetrics[store] = {
      FF: metrics.countPlan > 0 ? (metrics.countStock / metrics.countPlan * 100).toFixed(2) + '%' : '0.00%',
      FA: metrics.countArticlePlan > 0 ? (metrics.countArticle / metrics.countArticlePlan * 100).toFixed(2) + '%' : '0.00%',
      FS: metrics.totalPlan > 0 ? (metrics.totalStock / metrics.totalPlan * 100).toFixed(2) + '%' : '0.00%',
      ST: sellThroughMetrics[store] || '0.00%'
    };
  }
  
  console.log('Final Metrics:', calculatedMetrics);
  return calculatedMetrics;
}

/**
 * Update metric section di report sheet
 */
function updateMetricSection(sheet, metricType, dateString, calculatedMetrics, startRow) {
  const headerRow = sheet.getRange(startRow, 1, 1, sheet.getMaxColumns()).getValues()[0];
  let lastFilledColumn = 1;
  
  for (let i = 1; i < headerRow.length; i++) {
    if (headerRow[i] !== '' && headerRow[i] !== null) {
      lastFilledColumn = i + 1;
    }
  }
  
  const newColumn = lastFilledColumn + 1;
  
  sheet.getRange(startRow, newColumn).setValue(dateString);
  sheet.getRange(startRow + 1, newColumn).setValue(metricType);
  
  const stores = [
    'Zuma Matos',
    'Zuma Galaxy Mall',
    'Zuma Tunjungan Plaza',
    'ZUMA PTC',
    'Zuma Icon Gresik',
    'Zuma Lippo Sidoarjo',
    'Zuma Lippo Batu',
    'Zuma Royal Plaza',
    'Zuma City Of Tomorrow Mall',
    'Zuma Sunrise Mall',
    'Zuma Mall Olympic Garden'
  ];
  
  for (let i = 0; i < stores.length; i++) {
    const storeName = stores[i];
    const rowIndex = startRow + 2 + i;
    
    if (sheet.getRange(rowIndex, 1).getValue() === '') {
      sheet.getRange(rowIndex, 1).setValue(storeName);
    }
    
    if (calculatedMetrics[storeName]) {
      sheet.getRange(rowIndex, newColumn).setValue(calculatedMetrics[storeName][metricType]);
    } else {
      sheet.getRange(rowIndex, newColumn).setValue('0.00%');
    }
  }
}

/**
 * Remove trigger daily report
 */
function removeDailyTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  let removed = 0;
  
  triggers.forEach(trigger => {
    if (trigger.getHandlerFunction() === 'updateDailyReport') {
      ScriptApp.deleteTrigger(trigger);
      removed++;
    }
  });
  
  if (removed > 0) {
    SpreadsheetApp.getActiveSpreadsheet().toast(
      'Auto update telah dimatikan', 
      '🛑 Trigger Dihapus', 
      3
    );
  }
}

/**
 * Clear section helper
 */
function clearSection(sheet, startRow, endRow) {
  const lastColumn = sheet.getLastColumn();
  if (lastColumn > 1) {
    sheet.getRange(startRow, 2, 2, lastColumn - 1).clear();
    if (endRow > startRow + 1) {
      sheet.getRange(startRow + 2, 2, endRow - startRow - 1, lastColumn - 1).clear();
    }
  }
}
